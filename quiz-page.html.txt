        const display = document.getElementById('clock-display');
        
        setInterval(() => {
            if (timeRemaining <= 0) return;
            timeRemaining--;
            let minutes = Math.floor(timeRemaining / 60);
            let seconds = timeRemaining % 60;
            display.innerText = `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;
