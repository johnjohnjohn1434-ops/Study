        import { getDatabase, ref, push } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-database.js";

            apiKey: "AIzaSyAn7E7olxRpT_dBvQCL2K5nx-slpuTb1e8",
            authDomain: "study-327d0.firebaseapp.com",
            databaseURL: "https://study-327d0-default-rtdb.firebaseio.com",
            projectId: "study-327d0",
            storageBucket: "study-327d0.firebasestorage.app",
            messagingSenderId: "718476771682",
            appId: "1:718476771682:web:154c4253a4a68cc2aeec3b"
        };

        const app = initializeApp(firebaseConfig);
        const database = getDatabase(app);

        // Core dynamic action push sequence framework configuration script
        document.getElementById('upload-action-trigger').addEventListener('click', () => {
            const classVal = document.getElementById('material-class').value;
            const subjectVal = document.getElementById('material-subject').value;
            const chapterVal = document.getElementById('material-chapter').value;
            const typeVal = document.getElementById('material-type').value;
            const urlVal = document.getElementById('material-url').value;
            const descVal = document.getElementById('material-desc').value;

            if(!chapterVal || !urlVal) {
                alert("Please fill out the Chapter field and the Resource Link URL endpoint properly.");
                return;
            }

            // Reference mapping path tree config for database storage rows routing structure rules
            const materialsRef = ref(database, 'study_materials');
            
            const payload = {
                targetClass: classVal,
                subject: subjectVal,
                chapter: chapterVal,
                contentType: typeVal,
                resourceLink: urlVal,
                description: descVal,
                timestamp: Date.now()
            };

            push(materialsRef, payload)
                .then(() => {
                    alert("Data uploaded successfully to Firebase! Students can view this content dynamically.");
                    // Reset input elements
                    document.getElementById('material-chapter').value = '';
                    document.getElementById('material-url').value = '';
                    document.getElementById('material-desc').value = '';
                })
                .catch((error) => {
                    alert("Error inserting entry node mapping row row to system: " + error.message);
                });
        });
